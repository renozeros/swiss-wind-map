    <script>

        // Karte zentriert auf Ostschweiz / Rheintal / Bodensee

        const map = L.map('map').setView([47.38, 9.38], 10);

        

        L.tileLayer('https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png', {

            attribution: '&copy; OpenStreetMap'

        }).addTo(map);


        let markerLayer = L.layerGroup().addTo(map);


        async function updateMap() {

            try {

                // WICHTIG: Ersetze 'DEINE_RENDER_URL_HIER' durch deine echte Render-Webadresse nach Schritt 2!

                // Beispiel: 'https://onrender.com'

                const response = await fetch('DEINE_RENDER_URL_HIER/api/wind');

                const stations = await response.json();

                

                markerLayer.clearLayers();


                stations.forEach(station => {

                    const sourceClass = "source-" + station.source.split(" ")[0];

                    

                    const customIcon = L.divIcon({

                        className: `wind-marker ${sourceClass}`,

                        html: station.speed,

                        iconSize: [34, 34]

                    });


                    const popupId = `chart-${station.id}`;

                    const popupHTML = `

                        <div style="font-family: sans-serif;">

                            <strong style="font-size:13px;">${station.name}</strong><br>

                            <span style="font-size:11px; color:#666;">Quelle: ${station.source}</span><br>

                            <span style="font-size:12px;"><b>Aktuell:</b> ${station.speed} km/h (${station.direction}°)</span>

                            <div class="chart-container">

                                <canvas id="${popupId}"></canvas>

                            </div>

                        </div>

                    `;


                    const marker = L.marker([station.lat, station.lon], { icon: customIcon })

                        .addTo(markerLayer)

                        .bindPopup(popupHTML);


                    marker.on('popupopen', () => {

                        createPopupChart(popupId, station.history);

                    });

                });


                document.getElementById('status').innerHTML = `🟢 Live: ${stations.length} Stationen aktiv.`;

            } catch (error) {

                document.getElementById('status').innerHTML = "🔴 Fehler beim Laden der Daten oder Server schläft.";

            }

        }


        function createPopupChart(canvasId, historyData) {

            const ctx = document.getElementById(canvasId).getContext('2d');

            

            if (!historyData || historyData.length === 0) {

                ctx.font = "12px Arial";

                ctx.fillText("Keine Verlaufsdaten verfügbar", 10, 50);

                return;

            }


            const labels = historyData.map(h => h.time);

            const speeds = historyData.map(h => h.speed);


            new Chart(ctx, {

                type: 'line',

                data: {

                    labels: labels,

                    datasets: [{

                        label: 'km/h',

                        data: speeds,

                        borderColor: '#007bff',

                        backgroundColor: 'rgba(0, 123, 255, 0.1)',

                        borderWidth: 2,

                        tension: 0.3,

                        pointRadius: 2

                    }]

                },

                options: {

                    responsive: true,

                    maintainAspectRatio: false,

                    plugins: { legend: { display: false } },

                    scales: {

                        x: { grid: { display: false }, ticks: { font: { size: 9 } } },

                        y: { beginAtZero: true, ticks: { font: { size: 9 } } }

                    }

                }

            });

        }


        updateMap();

        setInterval(updateMap, 60000); // Aktualisierung im Minutentakt

    </script>
